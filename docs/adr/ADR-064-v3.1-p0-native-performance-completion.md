export class AttentionService {
  private useNative: boolean = true; // Enable native by default

  constructor(config?: AttentionConfig) {
    this.useNative = config?.useNative ?? true;

        this.useNative = false;
      }
    }
  }

  /**
   * Apply Flash Attention (7.47x faster than standard attention)
   */
  async applyFlashAttention(
    query: number[],
    keys: number[][],
    values: number[][],
    options?: { headCount?: number; dropoutRate?: number }
  ): Promise<number[]> {
    if (!this.useNative) {
      return this.applyAttentionJS(query, keys, values);
    }

    const queryBuffer = new Float32Array(query);
    const keysBuffer = new Float32Array(keys.flat());
    const valuesBuffer = new Float32Array(values.flat());

    // Native Flash Attention call
    const result = flashAttention(
      queryBuffer,
      keysBuffer,
      valuesBuffer,
      {
        numHeads: options?.headCount ?? 8,
        dropout: options?.dropoutRate ?? 0.1,
        seqLength: keys.length,
        dimPerHead: query.length / (options?.headCount ?? 8)
      }
    );

    return Array.from(result);
  }

  /**
   * Multi-Head Attention (5x better relevance)
   */
  async applyMultiHeadAttention(
    query: number[],
    context: number[][],
    numHeads: number = 8
  ): Promise<{ attention: number[]; weights: number[][] }> {
    if (!this.useNative) {
      return this.applyMultiHeadJS(query, context, numHeads);
    }

    const result = multiHeadAttention(
      new Float32Array(query),
      new Float32Array(context.flat()),
      { numHeads, contextSize: context.length }

    return {
      attention: Array.from(result.output),
      weights: this.reshapeWeights(result.weights, context.length, numHeads)

import { AttentionService } from './AttentionService.js';

export class WASMVectorSearch {
  private attentionService: AttentionService;

  constructor() {
    this.attentionService = new AttentionService({ useNative: true });
  }

  async search(
    query: number[],
    vectors: Array<{ id: string; vector: number[]; metadata?: any }>,
    topK: number = 10,
    useAttention: boolean = true
  ): Promise<Array<{ id: string; score: number; metadata?: any }>> {
    if (!useAttention) {
      return this.searchBasic(query, vectors, topK);
    }

    // Apply Flash Attention for relevance boost
    const keys = vectors.map(v => v.vector);
    const values = vectors.map(v => v.vector);

    const attentionScores = await this.attentionService.applyFlashAttention(
      query,
      keys,
      values,
      { headCount: 8 }
    );

    // Combine with cosine similarity
    const results = vectors.map((v, i) => ({
      id: v.id,
      score: this.computeScore(query, v.vector, attentionScores[i]),
      metadata: v.metadata
    }));

    return results
      .sort((a, b) => b.score - a.score)
      .slice(0, topK);
  }

  private computeScore(query: number[], vector: number[], attentionWeight: number): number {
    const cosineSim = this.cosineSimilarity(query, vector);
    return cosineSim * 0.7 + attentionWeight * 0.3; // Weighted combination
  }
}

export class QUICConnectionPool {
  private pools: Map<string, QUICConnection[]> = new Map();
  private maxPoolSize: number = 10;

  async getConnection(endpoint: string): Promise<QUICConnection> {
    const pool = this.pools.get(endpoint) || [];

    // Try to reuse existing connection
    for (const conn of pool) {
      if (!conn.isBusy()) {
        return conn;
      }
    }

    // Create new connection if pool not full
    if (pool.length < this.maxPoolSize) {
      const conn = await this.createConnection(endpoint);
      pool.push(conn);
      this.pools.set(endpoint, pool);
      return conn;
    }

    // Wait for available connection
    return this.waitForAvailableConnection(endpoint);
  }

  private async createConnection(endpoint: string): Promise<QUICConnection> {
    const conn = new QUICConnection({
      endpoint,
      enableZeroRTT: true,
      enableMultipath: true,
      congestionControl: 'bbr'
    });
    await conn.connect();
    return conn;
  }
}

export class QUICStreamManager {
  private streams: Map<number, QUICStream> = new Map();

  async sendMultiple(messages: Array<{ data: Buffer; priority: number }>): Promise<void> {
    // Create streams for each message
    const streamPromises = messages.map(async (msg, i) => {
      const stream = await this.createStream(i, msg.priority);
      await stream.send(msg.data);
    });

    // Send all messages concurrently
    await Promise.all(streamPromises);
  }

  private async createStream(id: number, priority: number): Promise<QUICStream> {
    const stream = new QUICStream({ id, priority });
    this.streams.set(id, stream);
    return stream;
  }
}

export interface ModelCost {
  modelId: string;
  inputCostPer1K: number;
  outputCostPer1K: number;
  latency: number;
  qualityScore: number;
}

export class CostOptimizerService {
  private costs: Map<string, ModelCost> = new Map();
  private budgetLimit: number = 100; // $100/month
  private currentSpend: number = 0;

  constructor() {
    this.initializeModelCosts();
  }

  private initializeModelCosts(): void {
    this.costs.set('claude-opus-4', {
      modelId: 'claude-opus-4',
      inputCostPer1K: 0.015,
      outputCostPer1K: 0.075,
      latency: 2000,
      qualityScore: 95
    });

    this.costs.set('claude-sonnet-4', {
      modelId: 'claude-sonnet-4',
      inputCostPer1K: 0.003,
      outputCostPer1K: 0.015,
      latency: 1000,
      qualityScore: 90
    });

    this.costs.set('claude-haiku-4', {
      modelId: 'claude-haiku-4',
      inputCostPer1K: 0.00025,
      outputCostPer1K: 0.00125,
      latency: 500,
      qualityScore: 80
    });

    this.costs.set('agent-booster', {
      modelId: 'agent-booster',
      inputCostPer1K: 0,
      outputCostPer1K: 0,
      latency: 1,
      qualityScore: 70
    });
  }

  /**
   * Select optimal model based on task complexity, budget, and latency requirements
   */
  selectOptimalModel(task: {
    complexity: number;      // 0-100
    maxLatency?: number;     // ms
    minQuality?: number;     // 0-100
    inputTokens: number;
    outputTokens: number;
  }): { modelId: string; estimatedCost: number; reasoning: string } {
    // Check if Agent Booster can handle it (free + instant)
    if (task.complexity < 30) {
      return {
        modelId: 'agent-booster',
        estimatedCost: 0,
        reasoning: 'Simple task, using Agent Booster (free + 1ms latency)'
      };
    }

    // Filter models by requirements
    const candidates = Array.from(this.costs.values()).filter(model => {
      if (task.maxLatency && model.latency > task.maxLatency) return false;
      if (task.minQuality && model.qualityScore < task.minQuality) return false;
      return true;
    });

    // Calculate value score for each candidate
    const scored = candidates.map(model => {
      const inputCost = (task.inputTokens / 1000) * model.inputCostPer1K;
      const outputCost = (task.outputTokens / 1000) * model.outputCostPer1K;
      const totalCost = inputCost + outputCost;

      // Value = Quality / (Cost * Latency)
      const value = model.qualityScore / (totalCost * model.latency);

      return { model, totalCost, value };
    });

    // Select highest value model
    scored.sort((a, b) => b.value - a.value);
    const selected = scored[0];

    // Check budget
    if (this.currentSpend + selected.totalCost > this.budgetLimit) {
      return {
        modelId: 'claude-haiku-4',
        estimatedCost: selected.totalCost,
        reasoning: 'Budget limit exceeded, using cheapest option'
      };
    }

    return {
      modelId: selected.model.modelId,
      estimatedCost: selected.totalCost,
      reasoning: `Best value: ${selected.model.qualityScore} quality at $${selected.totalCost.toFixed(4)}`
    };
  }

  /**
   * Track actual spend
   */
  recordSpend(modelId: string, inputTokens: number, outputTokens: number): void {
    const model = this.costs.get(modelId);
    if (!model) return;

    const cost = (inputTokens / 1000) * model.inputCostPer1K +
                 (outputTokens / 1000) * model.outputCostPer1K;

    this.currentSpend += cost;
  }

  /**
   * Get spending report
   */
  getSpendingReport(): {
    totalSpend: number;
    budgetRemaining: number;
    utilizationPercent: number;
    savingsVsOpus: number;
  } {
    const opusCost = this.costs.get('claude-opus-4')!;
    const avgOpusCost = (opusCost.inputCostPer1K + opusCost.outputCostPer1K) / 2;
    const estimatedOpusSpend = this.currentSpend / 0.1; // Assume 10% of Opus cost

    return {
      totalSpend: this.currentSpend,
      budgetRemaining: this.budgetLimit - this.currentSpend,
      utilizationPercent: (this.currentSpend / this.budgetLimit) * 100,
      savingsVsOpus: estimatedOpusSpend - this.currentSpend
    };
  }

  /**
   * Set budget limit
   */
  setBudget(limit: number): void {
    this.budgetLimit = limit;
  }
}

1. **Complexity** - More moving parts to maintain
2. **Platform Dependencies** - Native bindings require compatible systems
3. **Testing Burden** - More integration tests needed

### Risks & Mitigations

| Risk | Mitigation |
|------|------------|
| Native bindings fail | Automatic fallback to JS implementations |
| Budget overrun | Hard limits with alerts at 80% utilization |
| QUIC compatibility | Graceful degradation to HTTP/2 |
| Performance regression | Comprehensive benchmarking suite |
